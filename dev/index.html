HttpClient.Request — Type
struct Request
    response::String
    status::Int
    headers::Dict{String, String}
end

Store status, response and headers of successful http request.

Example

julia> request = HttpClient.get("https://example.com");

julia> request.status
200
julia> typeof(request.response)
String
julia> length(request.response)

julia> request.headers

Example

url = "https://reqres.in/api/users/2"

HttpClient.get — Method
get(url; headers, query, interface, timeout, retries) -> Request

Perform http get request and return Request object.

Example

headers = [
    "Content-Type" => "application/json", 
    "User-Agent" => "http-julia"
]
request = HttpClient.get("https://reqbin.com/echo/get/json"; headers)
@test request.status == 200
@test request.response == "{\"success\":\"true\"}\n"

HttpClient.post — Method
post(url; headers, query, interface, timeout, retries, body) -> Request

Perform http post request and return Request object.

Example

Successful registration

url = "https://reqres.in/api/register/"
headers = ["User-Agent" => "http-julia", "Content-Type" => "application/json"]
body = """{
    "email": "eve.holt@reqres.in",
    "password": "pistol"
}"""
request = HttpClient.post(url; headers, body)

@test request.status == 200
@test request.response == "{\"id\":4,\"token\":\"QpwL5tke4Pnpja7X4\"}"

Unsuccessful registration

url = "https://reqres.in/api/register/"
headers = ["User-Agent" => "http-julia", "Content-Type" => "application/json"]
body = """{
    "email": "eve.holt@reqres.in"
}""" # remove password field
request = HttpClient.post(url; headers, body)

@test request.status == 400
@test request.response == "{\"error\":\"Missing password\"}"

Clickhouse

url = "https://play.clickhouse.com/"
query = Dict("user" => "explorer")
headers = Dict("Content-Type" => "application/json", "User-Agent" => "http-julia")
body = "show databases"
request = HttpClient.post(url; query, headers, body)
@test request.status == 200
julia> print(request.response)
blogs
default
git_clickhouse
mgbench
system

HttpClient.put — Method
put(url; headers, query, interface, timeout, retries, body) -> Request

Perform http put request and return Request object.

Example

url = "https://reqres.in/api/users/2"
headers = ["User-Agent" => "http-julia", "Content-Type" => "application/json"]
body = """{
    "name": "morpheus",
    "job": "zion resident"
}"""
request = HttpClient.put(url; headers, body)

@test request.status == 200
julia> print(request.response)
{"name":"morpheus","job":"zion resident","updatedAt":"2023-08-25T12:17:32.283Z"}
